import React, { Component } from 'react';
import { createStore, combineReducers } from 'redux';
import { reducer as formReducer, submit } from 'redux-form';
import { Provider } from 'react-redux';
import { ReduxForm, ReduxFormField, TextInput, Modal } from '@athena/forge';

const rootReducer = combineReducers({
  form: formReducer,
});

const store = createStore(
  rootReducer
);

class ModalForm extends Component {

  constructor(props) {
    super(props);
  }

  render(){
    return (
      <div style={{ height: '440px' }}>
        <Provider store={store}>
          <Modal
            show={true}
            headerText="Delete Encounter"
            onPrimaryClick={() => store.dispatch(submit('redux-modal-demo'))}
            primaryButtonText="Delete"  onHide={() => { /* see Basic Modal demo */ }}
          >
            <p>Are you sure you want to delete this encounter? This action can not be undone.</p>
            <ReduxForm form="redux-modal-demo" includeSubmitButton={false} onSubmit={(values) => alert(JSON.stringify(values))} >
              <ReduxFormField
                id="reason"
                labelText="Reason"
                required
              />
              <ReduxFormField
                id="note"
                labelText="Note"
              />
            </ReduxForm>
          </Modal>
        </Provider>
      </div>
    );
  }

}

export default ModalForm;
