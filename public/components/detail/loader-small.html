import React, { Component } from 'react';
import { Loader, Button } from '@athena/forge';

export default class LoaderSmall extends Component {
  constructor(props) {
    super(props);
    this.state = { load: 'loading' };
    this.handleClick = this.handleClick.bind(this);
  }

  componentDidMount() {
    this.fakeLoadContent();
  }

  fakeLoadContent() {
    setTimeout(
      () => this.setState({ load: 'loaded' }),
      3000
    );
  }

  handleClick() {
    this.setState({ load: 'loading' });
    this.fakeLoadContent();
  }

  render() {
    return (
      <div className="fe_u_margin--large" style={{ border: '1px solid gray', height: '300px', width: '400px' }}>
        <Loader loading={ this.state.load === 'loading' } text="Loading demo content...">
          <div className="fe_u_padding--medium">
            <p>I am some content that is {this.state.load}</p>
            <p>I am some content that is {this.state.load}</p>
            <p>I am some content that is {this.state.load}</p>
            <Button text="Click me to reload content" onClick={this.handleClick} />
          </div>
        </Loader>
      </div>
    );
  }
}
