import React, { Component } from 'react';
import { createStore, combineReducers } from 'redux';
import { reducer as formReducer } from 'redux-form';
import { Provider } from 'react-redux';
import {
  Checkbox,
  DateInput,
  RadioGroup,
  ReadOnlyInput,
  ReduxForm,
  ReduxFormField,
  Select,
  TextInput,
  Textarea
} from '@athena/forge';
import moment from 'moment';

const rootReducer = combineReducers({
  form: formReducer,
});

const store = createStore(
  rootReducer
);

class FormUserData extends Component {

  constructor(props) {
    super(props);
    this.state = { formValues: '' };
    this.handleSubmit = this.handleSubmit.bind(this);
  }

  handleSubmit(values){
    this.setState({ formValues: JSON.stringify(values, null, 2) });
  }

  render(){
    return (
      <Provider store={store}>
        <div className="fe_u_padding--large">
          <ReduxForm form="redux-form-demo" onSubmit={this.handleSubmit}>

            <ReduxFormField
              id="name"
              labelText="Name"
              required
            />

            <ReduxFormField
              id="dob"
              labelText="Date of Birth"
              inputAs={DateInput}
              showCalendar={false}
              selectYearStart={moment.utc().year() - 100}
              selectYearEnd={moment.utc().year()}
              required
            />

            <ReduxFormField
              id="gender"
              inputAs={RadioGroup}
              labelText="Gender"
              options={[ "Male", "Female", "Other" ]}
              required
            />

            <ReduxFormField
              id="email"
              inputAs={TextInput}
              labelText="Email"
              constraints={{email:{message: 'Please enter a valid email'}}}
              required
            />

            <ReduxFormField
              id="occp"
              labelText="Occupation"
              inputAs={Select}
              required
            >
              <option value="job0">Job type 1</option>
              <option value="job1">Job type 2</option>
              <option value="job2">Job type 3</option>
              <option value="job3">Job type 4</option>
              <option value="job4">Job type 5</option>
            </ReduxFormField>

            <ReduxFormField
              id="notes"
              inputAs={Textarea}
              labelText="Notes"
            />

            <ReduxFormField
              id="terms"
              inputAs={Checkbox}
              labelText="Terms"
              description="I agree to terms"
              required
            />

            <ReduxFormField
              inputAs={ReadOnlyInput}
              labelText="Thank you"
              text="Thanks for entering your data!"
            />

          </ReduxForm>

          <div style={{
            boxSizing: 'border-box',
            fontFamily:'monospace',
            color:'dimgray',
            border:'1px solid gainsboro',
            background:'whitesmoke',
            padding:'1rem',
            borderRadius: '3px',
            width: '300px',
            minHeight: '62px'
          }}>
            <strong>FORM DATA:</strong><br />
            <code><pre>{this.state.formValues}</pre></code>
          </div>

        </div>
      </Provider>
    );
  }

}

export default FormUserData;
