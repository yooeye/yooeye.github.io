import React from 'react';
import { Select, Heading } from '@athena/forge';

export default class SelectStandalone extends React.Component {
  constructor(props) {
    super(props);

    this.state = { value: undefined };
  }

  render() {
    return (
      <div className="fe_u_margin--large">
        <Heading headingTag="h1" variant="section" text="Barber paradox" />
        <p>
          There is a town where the barber shaves all the men that
          don&#39;t shave themselves, and only them. Does the barber
          shave himself?
        </p>
        <Select onChange={(e) => {
          this.setState({ value: e.target.value });
        }}
        options={[
          { value: 'Y', text: 'The barber shaves himself.' },
          { value: 'N', text: 'The barber doesn\'t shave himself.' },
          { value: 'W', text: 'The barber doesn\'t shave herself.' },
        ]} />
        { this.state.value === 'Y' &&
          <p>But the barber shaves ONLY men that don&#39;t shave themselves!</p> }
        { this.state.value === 'N' &&
          <p>But the barber shaves ALL men that don&#39;t shave themselves!</p> }
        { this.state.value === 'W' &&
          <p>The barber is a woman? Genius!</p> }
      </div>
    );
  }
}
