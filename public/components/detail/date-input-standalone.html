import React from 'react';
import { DateInput } from '@athena/forge';
import { GridRow, GridCol } from '@athena/forge'


export default class DateInputStandalone extends React.Component {
  constructor(props) {
    super(props);

    this.state = {};
  }

  render() {
    return (
      <GridRow className="fe_u_padding--large">
        <GridCol width={{small: 12, medium: 6}}>
          <DateInput
            onDateChange={(event) => { this.setState({ date: event.target.dateValue })}}
            onChange={(event) => { this.setState({
              raw: event.target.rawValue,
              formatted: event.target.value,
            })}}
          />
        </GridCol>
        <GridCol width={{small: 12, medium: 6}}>
          <ul>
            <li>Raw value: {this.state.raw}</li>
            <li>Formatted value: {this.state.formatted}</li>
            <li>Date value: {this.state.date && DateInput.formatDate(this.state.date)}</li>
          </ul>
        </GridCol>
      </GridRow>
    );
  }
}
