import React from 'react';
import { DateInput, FormField } from '@athena/forge';
import moment from 'moment';

export default function DateInputField () {
  return (
    <div className="fe_u_padding--large">
      <FormField inputAs={DateInput} labelText="Date" />
      <FormField inputAs={DateInput} labelText="Date of Birth (don't automatically pop the calendar)"
        showCalendar={false}
        selectYearStart={moment.utc().year() - 100}
        selectYearEnd={moment.utc().year()}
      />
    </div>
  );
}
