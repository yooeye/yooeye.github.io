import React from 'react';
import { Select, Form, FormField } from '@athena/forge';

export default function SelectComplex () {
  return (
    <Form includeSubmitButton={false} className="fe_u_margin--large">
      <FormField
        inputAs={Select}
        labelWidth={4}
        labelText="A set of complex options"
        options={[
          { text: 'A', value: 1 },
          { text: 'B', value: 2 },
          { text: 'C', value: 3, disabled: true },
        ]} />
      <FormField
        inputAs={Select}
        labelWidth={4}
        labelText="Complex options as children"
      >
        <option value={1}>A</option>
        <option value={2}>B</option>
        <option value={3} disabled>C</option>
      </FormField>
    </Form>
  );
}
