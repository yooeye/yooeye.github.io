import React from 'react';
import { RadioGroup } from '@athena/forge';

const OPTIONS = [
  { value: 2, text: 'Two options' },
  { value: 3, text: 'Three options' },
  { value: 4, text: 'Four options' },
  { value: 5, text: 'Five options' },
  { value: 6, text: 'Six options (not allowed)', disabled: true },
];

export default class SelectStandalone extends React.Component {
  constructor(props) {
    super(props);

    this.handleChange = this.handleChange.bind(this);
    this.state = { value: 5 };
  }

  handleChange(event) {
    this.setState({ value: event.target.value });
  }

  render() {
    return (
      <div className="fe_u_margin--large">
        <RadioGroup
          id="meta-radio"
          onChange={this.handleChange}
          value={this.state.value}
          options={OPTIONS.slice(0, this.state.value)}
        />
      </div>
    );
  }
}
