import React from 'react';
import { Table, Heading } from '@athena/forge';
import faker from 'faker';

const procedures = [
  'Allergy shots',
  'Electrocardiogram',
  'Joint aspiration',
  'Pneumococcal Vaccination',
  'Tattoo Removal',
  'Complete Blood Count',
];
const data = procedures.map((procedure, index) => {
  return {
    id: index,
    procedure: procedure,
    patient: faker.name.findName(),
    cost: Number(faker.finance.amount(50, 150)),
  };
});


export default function TableDemo() {
  return (
    <div className="fe_u_margin--large" >
      <Heading variant="section" text="Billable Procedures"/>
      <Table
        rows={data}
        columns={[
          {
            key: 'procedure',
            displayName: 'Procedure',
          },
          {
            key: 'patient',
            displayName: 'Patient',
          },
          {
            key: 'cost',
            displayName: 'Cost (USD)',
            numeric: true,
            template: (num) =>
              num.toLocaleString('en-US', { style: 'currency', currency: 'USD' }),
          },
        ]}
      />
    </div>
  );
}
