import React, { Component } from 'react';
import { Modal, Button } from '@athena/forge';
import loremIpsum from 'lorem-ipsum';

class ModalToggle extends Component {
  constructor(props) {
    super(props);

    this.state = { shown: props.shownByDefault };

    this.showModal = this.showModal.bind(this);
    this.hideModal = this.hideModal.bind(this);
    this.completeModal = this.completeModal.bind(this);
  }

  showModal () {
    this.setState({ shown: true });
  }

  hideModal () {
    this.setState({ shown: false });
  }

  completeModal () {
    this.setState({ shown: false });
  }

  render() {
    return (
      <div>
        <Button text="Open Modal" onClick={this.showModal} />
        <Modal show={this.state.shown} headerText="Multiple User Conflict" onHide={this.hideModal} onPrimaryClick={this.completeModal} disableClose={true}>
          <p>You made changes at the same time as another user. Your changes could not be saved.</p>
        </Modal>
      </div>
    );
  }
}

class ModalDisableClose extends Component {

  constructor(props) {
    super(props);
  }

  render(){

    return (
      <div style={{ height: '280px' }}>
        <p>{loremIpsum({ units: 'paragraphs', count: 1 })}</p>
        <p>{loremIpsum({ units: 'paragraphs', count: 1 })}</p>
        <p>{loremIpsum({ units: 'paragraphs', count: 1 })}</p>
        <ModalToggle shownByDefault={true} />
      </div>
    );
  }

}

export default ModalDisableClose;
