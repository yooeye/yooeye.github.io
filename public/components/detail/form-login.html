import React, { Component } from 'react';
import {
  Form,
  FormField
} from '@athena/forge';

class FormLogin extends Component {

  constructor(props) {
    super(props);
    this.state = { formValues: '' };
    this.handleSubmit = this.handleSubmit.bind(this);
  }

  handleSubmit(event){
    event.preventDefault();
    let values = {};
    Array.prototype.forEach.call(event.target.elements, function(el){
      if (el.value) {
        values[el.name] = el.value;
      }
    });
    this.setState({ formValues: JSON.stringify(values, null, 2) });
  }

  render(){
    return (
      <div className="fe_u_padding--large">
        <Form onSubmit={this.handleSubmit} buttonText="Login">

          <FormField
            id="username"
            labelText="Username"
          />

          <FormField
            id="password"
            labelText="Password"
          />

        </Form>

        <div style={{
          boxSizing: 'border-box',
          fontFamily:'monospace',
          color:'dimgray',
          border:'1px solid gainsboro',
          background:'whitesmoke',
          padding:'1rem',
          borderRadius: '3px',
          width: '300px',
          minHeight: '62px'
        }}>
          <strong>FORM DATA:</strong><br />
          <code><pre>{this.state.formValues}</pre></code>
        </div>

      </div>
    );
  }

}

export default FormLogin;
